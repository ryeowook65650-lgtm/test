const SHEET_NAME = '工作表1'; // 如果你的工作表名稱不同，請修改這裡 function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const action = e.parameter.action; if (action === 'add') { sheet.appendRow([ e.parameter.id, e.parameter.date, e.parameter.cat, e.parameter.name, e.parameter.price, e.parameter.user ]); return ContentService.createTextOutput(JSON.stringify({"status": "success"})).setMimeType(ContentService.MimeType.JSON); } else if (action === 'delete') { const idToDelete = e.parameter.id; const data = sheet.getDataRange().getValues(); for (let i = data.length - 1; i >= 0; i--) { // 🌟 在這裡加上 String() 保護，避免空白列導致程式錯誤 🌟 if (String(data[i][0]) === String(idToDelete)) { sheet.deleteRow(i + 1); return ContentService.createTextOutput(JSON.stringify({"status": "deleted"})).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({"status": "not_found"})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const action = e.parameter.action; if (action === 'read') { const data = sheet.getDataRange().getValues(); let result = []; for (let i = 1; i < data.length; i++) { let row = data[i]; if (row[0]) { result.push({ id: row[0], date: row[1], cat: row[2], name: row[3], price: row[4], user: row[5] || "" }); } } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } }
